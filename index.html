function showBingoEffect() { const effect = document.getElementById('bingoEffect'); effect.style.display = 'block'; // ← 明示的に表示する！ effect.style.animation = 'pop 0.6s ease'; // ← アニメーションも直接指定 setTimeout(() => { effect.style.display = 'none'; // ← 3秒後に消す effect.style.animation = ''; // ← アニメリセット }, 3000); }
